#### 冒泡排序

```javascript
// 冒泡排序
function bubbleSort(arr) {
    // 非数组或者数据里只有一个元素，直接返回
    if (arr.length <= 1) {
        return arr;
    }
    let exchange = true;
    // 每一轮的相邻元素比较，会将最大值后移，所以下一轮的比较元素，会减少i位
    // 数组中相邻元素，两两对比，最大的数字逐渐移到最后
    for (let i = 0; exchange && i < arr.length - 1; i++) {
        exchange = false;
        for (let j = 0; j < arr.length - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                exchange = true;
                let temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
    return arr;
}
```
